public class TestCompileFile {
    Node next;
    Object value;
    public TestCompileFile(int value) {
        this.value =  value;
    }

    public TestCompileFile(String value) {
        this.value =  value;
    }
    public void setNext(Node next) {
        this.next = next;
    }

    public Object getValue(NodeType nodeType) {
        switch (nodeType){
            case integer:
                return (int)value;
            case String:
                return (String)value;
        }
        return null;
    }

    public Node getNext() {
        return next;
    }

}
